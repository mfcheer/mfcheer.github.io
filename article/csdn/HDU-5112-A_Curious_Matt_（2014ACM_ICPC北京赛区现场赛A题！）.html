#include<utility>
#include<malloc.h>
#include<stdexcept>

using namespace std;

struct q
{
    int t ;
    int dis;
}p[10005];

int t ;
int n;

bool cmp (q a,q b)
{
    return a.t < b.t;
}

int k=1;

int main()
{
    scanf("%d",&t);
    while (t--)
    {
        scanf("%d",&n);
        for(int i=1;i<=n;i++)
            scanf("%d %d",&p[i].t,&p[i].dis);
        sort(p+1,p+1+n,cmp);

        double ans = (p[2].dis - p[1].dis) * 1.0 / ( p[2].t - p[1].t);

        for(int i=3;i<=n;i++)
        {
            ans = max (ans ,double ( abs( (p[i].dis-p[i-1].dis) )*1.0 / (p[i].t - p[i-1].t) ) );
        }
        printf("Case #%d: %.2lf\n",k++,ans);
    }
    return 0;
}
