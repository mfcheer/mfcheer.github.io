LIGHTOJ 1019 – BRUSH (V) 【最短路】

using namespace std;

int t, cases = 1;
int n, m, f;
const int MAXN = 110;
const int INF = 10000000;
int map[MAXN][MAXN];
int a, b, c;
bool vis[MAXN];
int lowcost[MAXN];

void dijkstra(int cost[][MAXN], int lowcost[], int n, int beg)
{
    for (int i = 0; i < n; i++)
    {
        vis[i] = false;
        lowcost[i] = cost[beg][i];
    }
    lowcost[beg] = 0;
    vis[beg] = true;
    for (int j = 0; j < n; j++)
    {
        int k = -1;
        int MIN = INF;
        for (int i = 0; i < n; i++)
            if (!vis[i] && lowcost[i] < MIN)
            {
                MIN = lowcost[i];
                k = i;
            }
        if (k == -1) break;
        vis[k] = true;

        for (int i = 0; i < n; i++)
        {
            if (!vis[i] && lowcost[k] + cost[k][i] < lowcost[i])
            {
                lowcost[i] = lowcost[k] + cost[k][i];
            }
        }
    }
}

int main()
{
    scanf("%d",&t);
    while (t--)
    {
        getchar();
        scanf("%d%d",&n,&m);
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
            {
                if (i == j) map[i][i] = 0;
                else map[i][j] = map[j][i] = INF;
            }
        while (m--)
        {
            scanf("%d%d%d", &a, &b, &c);
            if (map[a-1][b-1] > c)
            map[a-1][b-1] = map[b-1][a-1] = c;
        }
        dijkstra(map, lowcost, n, 0);
        printf("Case %d: ",cases++);
        if (lowcost[n - 1] != INF) printf("%d\n",lowcost[n-1]);
        else puts("Impossible");
    }
    return 0;
}
