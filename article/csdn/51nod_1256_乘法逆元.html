using namespace std;

long long ext_gcd(long long a, long long b, long long &x, long long &y)
{
    if (a == 0 && b == 0) return -1;
    if (b == 0)
    {
        x = 1;
        y = 0;
        return a;
    }
    long long d = ext_gcd(b, a%b, y, x);
    y -= a / b*x;
    return d;
}

//求逆元素
// ax=1(mod n)
long long mod_reverse(long long a, long long n)
{
    long long x, y;
    long long d = ext_gcd(a, n, x, y);
    if (d == 1)
        return (x%n + n) % n;
    else
        return -1;
}

int main()
{
    long long a, b;
    while (cin >> a >> b)
    {
        int ans = mod_reverse(a, b);
        cout << ans << endl;
    }
    return 0;
}
