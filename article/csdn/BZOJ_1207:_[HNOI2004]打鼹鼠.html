动态规划 当前的每步必是由之前的某一步得来的

代码：

#include <iostream>
#include <algorithm>
#include <set>
#include <map>
#include <string.h>
#include <queue>
#include <sstream>
#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <string>

using namespace std;

int n,m;

struct Node
{
    int x,y;
    int t;
}a[10100];

int f[10010];

int main()
{
    while (scanf("%d%d",&n,&m) !=EOF)
    {
        for(int i=1;i<=m;i++)
            {
                f[i]=1;
                scanf("%d%d%d",&a[i].t,&a[i].x,&a[i].y);
            }
        int ans = 1;
        for(int i=1;i<=m;i++)
        {
            for(int j=i+1;j<=m;j++)
            {
                if (abs(a[i].x - a[j].x) + abs(a[i].y - a[j].y) <= a[j].t - a[i].t)
                    f[j] = max(f[j],f[i] + 1);
                ans = max(ans, f[j]);
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
