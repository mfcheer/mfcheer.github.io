#include<stdio.h>
#include<stdlib.h>
#include<malloc.h>
#include<string.h>
#include<string.h>
#include<algorithm>

using namespace std;

int a[10005];

int main()
{
    int  n, caseno = 0, cases;
    scanf("%d", &cases);
    while( cases-- ) {
        scanf("%d %d %d %d %d %d %d", &a[0], &a[1],&a[2],&a[3],&a[4],&a[5], &n);
        for(i=0;i<=5;i++)
        a[i]=a[i]%10000007;
        for (int i=6;i<=n;i++)
            a[i]=(a[i-1]+a[i-2]+a[i-3]+a[i-4]+a[i-5]+a[i-6])% 10000007;
        printf("Case %d: %d\n", ++caseno, a[n]);
    }
    return 0;
}
