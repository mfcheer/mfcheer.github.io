void dfs(int i,int j )
{
    if(!visit[i][j] && a[i][j]=='@')
    {
        visit[i][j]=1;
        dfs(i-1,j-1);dfs(i-1,j);dfs(i-1,j+1);
        dfs(i,j-1);             dfs(i,j+1);
        dfs(i+1,j-1);dfs(i+1,j);dfs(i+1,j+1);
    }
}
