poj 3320 Jessica’s Reading Problem 【尺取法】

版权声明：转载请注明出处。 https://blog.csdn.net/u014427196/article/details/46380533

题目：链接： http://poj.org/problem?id=3320

题意：求最短的连续区间使得全部的知识点都学到；

尺取法，不断推进区间的起点和终点，求解。

代码：

#include <stdio.h>
#include <iostream>
#include <string.h>
#include <algorithm>
#include <math.h>
#include <ctype.h>
#include <time.h>
#include <queue>
#include <iterator>
#include <vector>
#include <set>
#include <string>
#include <map>

using namespace std;

int n;
int p[1000010];

int main()
{
    while(scanf("%d", &n)!=EOF)
    {
        set<int> SET;
        map<int, int> MAP;

        for (int i = 0; i < n; i++)
            scanf("%d", &p[i]);
        for (int i = 0; i < n; i++)
            SET.insert(p[i]);

        int len = SET.size();
        int ans = n;
        int s , t , sum;
        s = t = sum = 0;
        while (1)
        {
            while (t < n && sum < len)
            {
                if (MAP[p[t++]]++ == 0)
                    sum++;
            }
            if (sum < len) break;
            ans = min(ans,t-s);
            if (--MAP[p[s++]] == 0)
                sum--;
        }
        printf("%d\n",ans);
    }
    return 0;
}
